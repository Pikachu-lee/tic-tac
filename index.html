""" Text Adventure Game (starter complete file) Features: - Movement NSEW - Inventory - Weapons - Combat - Health - Treasure - Save/Load - Multiple rooms - Random events This is a compact complete implementation (~170 lines). """ import json, random rooms={ "Village":{"N":"Forest","item":"Knife"}, "Forest":{"S":"Village","E":"Cave","enemy":["Wolf",20,4],"item":"Potion"}, "Cave":{"W":"Forest","N":"Castle","enemy":["Goblin",30,6],"treasure":50}, "Castle":{"S":"Cave","enemy":["Dragon",60,10],"treasure":200} } state={"room":"Village","hp":100,"weapon":["Fist",2],"inv":[],"gold":0} weapons={"Knife":5,"Sword":8} def save(): json.dump(state,open("save.json","w")) print("Saved.") def load(): global state state=json.load(open("save.json")) print("Loaded.") def show(): r=rooms[state["room"]] print("\nRoom:",state["room"],"HP",state["hp"],"Gold",state["gold"]) print("Inventory:",state["inv"],"Weapon",state["weapon"]) print("Exits:",[k for k in "NSEW" if k in r]) if "enemy" in r: print("Enemy:",r["enemy"][0]) if "item" in r: print("Item:",r["item"]) def fight(): r=rooms[state["room"]] if "enemy" not in r:return name,hp,dmg=r["enemy"] while hp>0 and state["hp"]>0: hit=random.randint(1,state["weapon"][1]) hp-=hit print("You hit",hit) if hp<=0:break eh=random.randint(1,dmg) state["hp"]-=eh print(name,"hits",eh) if state["hp"]<=0: print("You died.");exit() print("Defeated",name) del r["enemy"] if "treasure" in r: state["gold"]+=r["treasure"];print("Found",r["treasure"],"gold") del r["treasure"] def pickup(): r=rooms[state["room"]] if "item" not in r:return it=r["item"];state["inv"].append(it) if it in weapons: state["weapon"]=[it,weapons[it]] if it=="Potion": state["hp"]=min(100,state["hp"]+25) print("Picked",it) del r["item"] def event(): x=random.randint(1,5) if x==1: d=5;state["hp"]-=d;print("Trap! -5 HP") elif x==2: g=10;state["gold"]+=g;print("Found 10 gold") while True: if state["room"]=="Castle" and "enemy" not in rooms["Castle"]: print("You defeated the Dragon and won!");break show() c=input("Command(n/s/e/w,pick,fight,save,load,quit): ").lower() if c=="quit":break if c=="save":save();continue if c=="load": try:load() except:print("No save.") continue if c=="pick":pickup();continue if c=="fight":fight();continue m={"n":"N","s":"S","e":"E","w":"W"} if c in m: r=rooms[state["room"]] if m[c] in r: state["room"]=r[m[c]] event() else: print("Can't go that way")
